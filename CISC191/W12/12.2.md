import java.util.Scanner;

public class MergeSortActivity {

static int comparisons = 0;

public static void printArray(int[] numbers) {

for (int i = 0; i < numbers.length; i++) {

if (i > 0) {
  System.out.print(" ");
}

System.out.print(numbers[i]);
}

System.out.println();
}

//merge step


public static void merge(int[] numbers, int i, int j, int k) {
int mergedSize = k - i + 1;
int[] merged = new int[mergedSize];
int mergedPos = 0;
int leftPos = i;
int rightPos = j + 1;


while (leftPos <= j && rightPos <= k) {

comparisons++;

if (numbers[leftPos] < numbers[rightPos]) {
merged [mergedPos] = numbers[leftPos];
leftPos++;
} else {
merged[mergedPos] = numbers[rightPos];
rightPos++;
}
mergedPos++;
}

while (leftPos <= j) {
merged[mergedPos] =numbers[leftPos];
leftPos++;
mergedPos++;
}

while (rightPos <= k) {
merged[mergedPos] = numbers[rightPos];
rightPos++;
mergedPos++;
}

for ( int m = 0; m < mergedSize; m++) {
 numbers[i + m] = merged[m];
}
}


public static void mergeSort(int[] numbers, int i , int k) {
if (i < k) {
int j = (i + k) / 2;

mergeSort(numbers, i, j);
mergeSort(numbers, j + 1, k);

merge(numbers, i, j, k);
}
}

public static void main(String[] args) {
Scanner sc = new Scanner(System.in);

int n = sc.nextInt();

int[] numbers = new int[n];
for (int i = 0; i < n; i++) {
numbers[i] = sc.nextInt();
}

System.out.print("unsorted: ");
printArray(numbers);
System.out.println();

comparisons = 0;
if (n > 0) {

 mergeSort(numbers, 0, n - 1);
}

System.out.print("sorted:   ");
printArray(numbers);

System.out.println("comparisons: " + comparisons);

sc.close();
}
}
