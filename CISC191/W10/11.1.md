import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.text.NumberFormat;

public class SalaryCalculatorGUI extends JFrame {
private final JTextField wageField;
private final JTextField hoursField;
private final JLabel resultLabel;
private final JButton calcButton;

public SalaryCalculatorGUI() {
super("Yearly Salary Calculator");
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

JPanel panel = new JPanel(new GridBagLayout());
panel.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 12));
GridBagConstraints gbc = new GridBagConstraints();
gbc.insets = new Insets(6, 6, 6, 6);
gbc.fill = GridBagConstraints.HORIZONTAL;

JLabel wageLabel = new JLabel("Hourly wage ($):");
wageField = new JTextField(10);
wageField.setToolTipText("Enter your hourly pay, e.g., 22.50");

gbc.gridx = 0; gbc.gridy = 0; gbc.weightx = 0;
panel.add(wageLabel, gbc);
gbc.gridx = 1; gbc.gridy = 0; gbc.weightx = 1;
panel.add(wageField, gbc);

JLabel hoursLabel = new JLabel("Hours per week:");
hoursField = new JTextField(10);
hoursField.setToolTipText("Enter hours you work per week, e.g., 35");

gbc.gridx = 0; gbc.gridy = 1; gbc.weightx = 0;
panel.add(hoursLabel, gbc);
gbc.gridx = 1; gbc.gridy = 1; gbc.weightx = 1;
panel.add(hoursField, gbc);

calcButton = new JButton("Calculate Yearly Salary");
gbc.gridx = 0; gbc.gridy = 2; gbc.gridwidth = 2; gbc.weightx = 1;
panel.add(calcButton, gbc);

resultLabel = new JLabel("Yearly salary: -");
resultLabel.setFont(resultLabel.getFont().deriveFont(Font.BOLD, 16f));
gbc.gridx = 0; gbc.gridy = 3; gbc.gridwidth = 2;
panel.add(resultLabel, gbc);

ActionListener calculateAction = new ActionListener() {
@Override
public void actionPerformed(ActionEvent e) {
calculateAndDisplay();
}
};

calcButton.addActionListener(calculateAction);
wageField.addActionListener(calculateAction);
hoursField.addActionListener(calculateAction);

setContentPane(panel);
pack();
setLocationRelativeTo(null);
}


private void calculateAndDisplay() {
String wageText = wageField.getText().trim();
String hoursText = hoursField.getText().trim();

double wage, hours;

try {
wage = Double.parseDouble(wageText);
if (wage < 0) {
throw new IllegalArgumentException("Hourly wage cannot be negative.");
}
} catch (NumberFormatException ex) {
showError("Please enter a valid number for Hourly wage (e.g., 22.50).");
wageField.requestFocus();
wageField.selectAll();
return;
} catch (IllegalArgumentException ex) {
showError(ex.getMessage());
wageField.requestFocus();
wageField.selectAll();
return;
}

try {
hours = Double.parseDouble(hoursText);
if (hours < 0 || hours > 168) {
throw new IllegalArgumentException("Hours per week must be between 0 and 168.");
}
} catch (NumberFormatException ex) {
showError("Please enter a valid number for Hours per week (e.g., 35).");
hoursField.requestFocus();
hoursField.selectAll();
return;
} catch (IllegalArgumentException ex) {
showError(ex.getMessage());
hoursField.requestFocus();
hoursField.selectAll();
return;
}

double yearly = wage * hours * 52;

String formatted = NumberFormat.getCurrencyInstance().format(yearly);
resultLabel.setText("Yearly salary: " + formatted);
}

private void showError(String message) {
JOptionPane.showMessageDialog(
this,
message,
"Input Error",
JOptionPane.ERROR_MESSAGE
);
}

public static void main(String[] args) {
SwingUtilities.invokeLater(() -> new SalaryCalculatorGUI().setVisible(true));
}
}
