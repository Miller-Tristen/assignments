import javax.swing.*;
import javax.swing.text.NumberFormatter;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.text.DecimalFormat;
import java.text.NumberFormat;

public class DistanceConverterGUI extends JFrame {

private final JFormattedTextField milesField;
private final JLabel kmLabel;
private final JLabel mLabel;
private final JLabel ftLabel;
private final JButton calcButton;

private static final double KM_PER_MILE = 1.609344;
private static final double M_PER_MILE = 1609.344;
private static final double FT_PER_MILE = 5280.0;

public DistanceConverterGUI() {
super("Distance Converter (Miles -> km, m, ft)");
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

NumberFormat nf = NumberFormat.getNumberInstance();
if (nf instanceof DecimalFormat) {
((DecimalFormat) nf).applyPattern("#,##0.###");
}
NumberFormatter milesFormatter = new NumberFormatter(nf);
milesFormatter.setValueClass(Double.class);
milesFormatter.setMinimum(0.0);
milesFormatter.setAllowsInvalid(false);
milesFormatter.setCommitsOnValidEdit(true);

milesField = new JFormattedTextField(milesFormatter);
milesField.setColumns(10);
milesField.setToolTipText("Enter distance in miles (e.g., 3.5)");

JPanel panel = new JPanel(new GridBagLayout());
panel.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 12));
GridBagConstraints gbc = new GridBagConstraints();
gbc.insets = new Insets(6, 6, 6, 6);
gbc.fill = GridBagConstraints.HORIZONTAL;

gbc.gridx = 0; gbc.gridy = 0; gbc.weightx = 0;
panel.add(new JLabel("Miles:"), gbc);
gbc.gridx = 1; gbc.gridy = 0; gbc.weightx = 1;
panel.add(milesField, gbc);

calcButton = new JButton("Convert");
gbc.gridx = 0; gbc.gridy = 1; gbc.gridwidth = 2; gbc.weightx = 1;
panel.add(calcButton, gbc);

kmLabel = new JLabel("—");
mLabel  = new JLabel("—");
ftLabel = new JLabel("—");


Font f = kmLabel.getFont().deriveFont(Font.BOLD, 15f);
kmLabel.setFont(f);
mLabel.setFont(f);
ftLabel.setFont(f);

 gbc.gridwidth = 1;
gbc.gridx = 0; gbc.gridy = 2; gbc.weightx = 0;
panel.add(new JLabel("Kilometers:"), gbc);
gbc.gridx = 1; gbc.gridy = 2; gbc.weightx = 1;
panel.add(kmLabel, gbc);

gbc.gridx = 0; gbc.gridy = 3; gbc.weightx = 0;
panel.add(new JLabel("Meters:"), gbc);
gbc.gridx = 1; gbc.gridy = 3; gbc.weightx = 1;
panel.add(mLabel, gbc);

gbc.gridx = 0; gbc.gridy = 4; gbc.weightx = 0;
panel.add(new JLabel("Feet:"), gbc);
gbc.gridx = 1; gbc.gridy = 4; gbc.weightx = 1;
panel.add(ftLabel, gbc);

// event

 ActionListener convertAction = new ActionListener() {
@Override
public void actionPerformed(ActionEvent e) {
convertAndDisplay();
}
};
calcButton.addActionListener(convertAction);
milesField.addActionListener(convertAction);
getRootPane().setDefaultButton(calcButton);

setContentPane(panel);
pack();
setLocationRelativeTo(null);
}

private void convertAndDisplay() {
Object value = milesField.getValue();
if (value == null) {
showError("Please enter a valid non-negative number for miles.");
focusMiles();
return;
}

double miles;
try {
miles = ((Number) value).doubleValue();
if (miles < 0) {
throw new IllegalArgumentException("Miles cannot be negative.");
}
} catch (Exception ex) {
showError("Please enter a valid non-negative number for miles.");
focusMiles();
return;
}

double km = miles * KM_PER_MILE;
double m  = miles * M_PER_MILE;
double ft = miles * FT_PER_MILE;

// format output
DecimalFormat outFmt = new DecimalFormat("#,##0.###");
kmLabel.setText(outFmt.format(km) + " km");
mLabel.setText(outFmt.format(m) + " m");
ftLabel.setText(outFmt.format(ft) + " ft");
}
private void showError(String msg) {
JOptionPane.showMessageDialog(this, msg, "Input Error", JOptionPane.ERROR_MESSAGE);
}

private void focusMiles() {
SwingUtilities.invokeLater(() -> {
milesField.requestFocus();
milesField.selectAll();
});
}

public static void main(String[] args) {
SwingUtilities.invokeLater(() -> new DistanceConverterGUI().setVisible(true));
}
}
