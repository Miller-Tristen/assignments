import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.text.DecimalFormat;
import java.text.ParseException;

public class MilesToKmSpinnerGUI extends JFrame {

private final JSpinner milesSpinner;
private final JButton convertButton;
private final JLabel resultLabel;

private static final double KM_PER_MILE = 1.609344;

public MilesToKmSpinnerGUI() {

super("Miles → Kilometers (JSpinner)");
setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

SpinnerNumberModel model = new SpinnerNumberModel(0.0, 0.0, null, 0.1);
milesSpinner = new JSpinner(model);

 JSpinner.NumberEditor editor = new JSpinner.NumberEditor(milesSpinner, "#,##0.###");
milesSpinner.setEditor(editor);

convertButton = new JButton("Convert");
resultLabel = new JLabel("Kilometers: —");
resultLabel.setFont(resultLabel.getFont().deriveFont(Font.BOLD, 16f));

 JPanel panel = new JPanel(new GridBagLayout());
panel.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 12));
GridBagConstraints gbc = new GridBagConstraints();
gbc.insets = new Insets(6, 6, 6, 6);
gbc.fill = GridBagConstraints.HORIZONTAL;

gbc.gridx = 0; gbc.gridy = 0; gbc.weightx = 0;
panel.add(new JLabel("Miles:"), gbc);
gbc.gridx = 1; gbc.gridy = 0; gbc.weightx = 1;
panel.add(milesSpinner, gbc);

gbc.gridx = 0; gbc.gridy = 1; gbc.gridwidth = 2; gbc.weightx = 1;
panel.add(convertButton, gbc);

gbc.gridx = 0; gbc.gridy = 2; gbc.gridwidth = 2;
panel.add(resultLabel, gbc);

ActionListener convertAction = new ActionListener() {
@Override
public void actionPerformed(ActionEvent e) {
convertAndDisplay();
}
};
convertButton.addActionListener(convertAction);

getRootPane().setDefaultButton(convertButton);
setContentPane(panel);
pack();
setLocationRelativeTo(null);
}

private void convertAndDisplay() {
try {
milesSpinner.commitEdit();
} catch (ParseException ignored) {

}

Object value = milesSpinner.getValue()
if (!(value instanceof Number)) {
showError("Please enter a valid non-negative number for miles.");
focusSpinner();
return;
}

double miles = ((Number) value).doubleValue();
if (miles < 0) {
showError("Miles cannot be negative.");
focusSpinner();
return;
}

double km = miles * KM_PER_MILE;
DecimalFormat fmt = new DecimalFormat("#,##0.###");
resultLabel.setText("Kilometers: " + fmt.format(km) + " km");
}

private void showError(String msg) {
JOptionPane.showMessageDialog(this, msg, "Input Error", JOptionPane.ERROR_MESSAGE);
}

private void focusSpinner() {
SwingUtilities.invokeLater(() -> {
Component editor = milesSpinner.getEditor();
if (editor instanceof JSpinner.NumberEditor) {
JFormattedTextField tf = ((JSpinner.NumberEditor) editor).getTextField();
tf.requestFocus();
tf.selectAll();
} else {
milesSpinner.requestFocus();
}
});
}

public static void main(String[] args) {
SwingUtilities.invokeLater(() -> new MilesToKmSpinnerGUI().setVisible(true));
}
}
